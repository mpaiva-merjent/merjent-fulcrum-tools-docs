/**
 * @class
 * @memberof DataEventHelpers
 * @summary Standalone HTTP request class for Fulcrum REST & Query APIs
 * @description Wraps Fulcrum's `REQUEST()` function with consistent error handling,
 * JSON parsing, and auth headers. Supports query, fetch, update, batch, and external API calls.
 *
 * **Pattern:** Standalone class instantiated via factory `M.mkRequest()` or directly.
 *
 * @example <caption>Via factory method</caption>
 * const req = M.mkRequest()
 * req.query("SELECT * FROM \"My App\"", function(rows) {
 *   SETVALUE('count', rows.length)
 * })
 *
 * @example <caption>Fetch-modify-update</caption>
 * const req = M.mkRequest()
 * req.modifyRecord(recID, function(record) {
 *   record.record.form_values['a1b2'] = 'New Value'
 *   return record
 * }, function() {
 *   ALERT('Updated!')
 * })
 *
 * @example <caption>Batch update</caption>
 * const req = M.mkRequest()
 * req.batchUpdate(recordIDs, function(record) {
 *   record.record.form_values['c3d4'] = 'Batch Value'
 *   return record
 * }, { completeMsg: 'All records updated!' })
 *
 * @example <caption>External API</caption>
 * const req = M.mkRequest()
 * req.external({
 *   url: 'https://api.example.com/data',
 *   qs: { lat: LATITUDE(), lon: LONGITUDE() }
 * }, function(data) {
 *   SETVALUE('result', data.value)
 * })
 */
class MRequest {
  /**
   * @param {Object} [config={}] - Optional initial configuration
   * @param {Function} [config.onError] - Custom error handler overriding the default
   */
  constructor(config = {}) {
    this.onError = config.onError || MRequest._defaultErrorHandler;
  }

  // ═══════════════════════════════════════════════════════════════════════════
  //  PRIVATE INFRASTRUCTURE
  // ═══════════════════════════════════════════════════════════════════════════

  /**
   * @summary Default error handler
   * @description Checks for connection errors and alerts accordingly.
   * Fixes the original `checkConnection` bug where both branches used `!=`.
   * @param {Object} error - Error object from REQUEST
   * @static
   */
  static _defaultErrorHandler = (error) => {
    var msg = error && error.message ? error.message : '';
    if (msg.slice(0, 22) === 'Unable to resolve host') {
      ALERT('Unable to complete task without data connection');
    } else {
      ALERT('Error with request: ' + INSPECT(error));
    }
  };

  /**
   * @summary Central REQUEST wrapper
   * @description All public methods delegate here. Handles options building,
   * auth header injection, error handling, and JSON parsing.
   *
   * @param {Object} cfg
   * @param {string} cfg.url - Request URL
   * @param {string} [cfg.method='GET'] - HTTP method
   * @param {Object} [cfg.headers] - Merged with default auth header
   * @param {Object} [cfg.qs] - Query string parameters
   * @param {string} [cfg.body] - Request body (for PUT/POST)
   * @param {string} [cfg.accept] - Accept header shorthand
   * @param {string} [cfg.contentType] - Content-Type shorthand
   * @param {boolean} [cfg.parseJSON=true] - Auto-parse body as JSON
   * @param {Function} [cfg.onSuccess] - Called with (parsedData, response)
   * @param {Function} [cfg.onError] - Per-request error override
   */
  _request = (cfg) => {
    var headers = cfg.skipAuth ? {} : { 'X-ApiToken': mToken };
    if (cfg.accept) headers['Accept'] = cfg.accept;
    if (cfg.contentType) headers['Content-Type'] = cfg.contentType;
    if (cfg.headers) {
      for (var k in cfg.headers) {
        if (cfg.headers.hasOwnProperty(k)) headers[k] = cfg.headers[k];
      }
    }

    var options = { url: cfg.url, method: cfg.method || 'GET', headers: headers };
    if (cfg.qs) options.qs = cfg.qs;
    if (cfg.body) options.body = cfg.body;
    if (cfg.accept) options.accept = cfg.accept;

    var onError = cfg.onError || this.onError;

    REQUEST(options, function(error, response, body) {
      if (error) { onError(error); return; }
      if (!body) return;
      var data = cfg.parseJSON !== false ? JSON.parse(body) : body;
      if (cfg.onSuccess) cfg.onSuccess(data, response);
    });
  };

  // ═══════════════════════════════════════════════════════════════════════════
  //  PUBLIC API
  // ═══════════════════════════════════════════════════════════════════════════

  /**
   * @summary Execute Query API SQL request
   * @description Sends a SQL query to the Fulcrum Query API and returns the rows.
   *
   * @param {string} query - SQL query string
   * @param {Function} callback - Called with (rows)
   * @example
   * req.query("SELECT * FROM \"My App\" WHERE _status = 'active'", function(rows) {
   *   SETVALUE('count', rows.length)
   * })
   */
  query = (query, callback) => {
    this._request({
      url: 'https://api.fulcrumapp.com/api/v2/query/',
      qs: { 'format': 'json', 'q': query },
      contentType: 'application/json',
      onSuccess: function(data) {
        if (callback) callback(data.rows || []);
      }
    });
  };

  /**
   * @summary GET a single record by ID
   * @description Fetches a record from the Fulcrum REST API without triggering any update.
   *
   * @param {string} recID - _record_id
   * @param {Function} callback - Called with (record)
   * @example
   * req.fetchRecord(RECORDID(), function(record) {
   *   SETVALUE('created_by', record.record.created_by)
   * })
   */
  fetchRecord = (recID, callback) => {
    this._request({
      url: 'https://api.fulcrumapp.com/api/v2/records/' + recID + '.json',
      onSuccess: function(data) {
        if (callback && data) callback(data);
      }
    });
  };

  /**
   * @summary PUT (update) a single record
   * @description Sends a PUT request to update a record via the Fulcrum REST API.
   *
   * @param {string} recID - _record_id
   * @param {Object} record - Full record object
   * @param {Function} [callback] - Called with (data, response) on success
   * @example
   * req.putRecord(recID, modifiedRecord, function() {
   *   ALERT('Record updated!')
   * })
   */
  putRecord = (recID, record, callback) => {
    this._request({
      url: 'https://api.fulcrumapp.com/api/v2/records/' + recID + '.json',
      method: 'PUT',
      headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
      body: JSON.stringify(record),
      onSuccess: function(data, response) {
        if (callback) callback(data, response);
      }
    });
  };

  /**
   * @summary Fetch-modify-update a single record
   * @description Fetches a record, passes it to buildFn for modification, then PUTs the result.
   * If buildFn returns null/undefined, the update is skipped.
   *
   * @param {string} recID - _record_id
   * @param {Function} buildFn - Receives record object, must return modified record (or null to skip)
   * @param {Function} [callback] - Called after successful PUT
   * @example
   * req.modifyRecord(recID, function(record) {
   *   record.record.form_values['a1b2'] = 'Updated'
   *   return record
   * }, function() {
   *   ALERT('Done!')
   * })
   */
  modifyRecord = (recID, buildFn, callback) => {
    var self = this;
    this.fetchRecord(recID, function(record) {
      var modified = buildFn(record);
      if (!modified) return;
      self.putRecord(recID, modified, callback);
    });
  };

  /**
   * @summary Batch fetch-modify-update for multiple records
   * @description Iterates over an array of record IDs, applying modifyRecord to each.
   * Optionally alerts on completion.
   *
   * @param {string[]} recIDs - Array of _record_id
   * @param {Function} buildFn - Receives record, returns modified record or null to skip
   * @param {Object} [opts={}]
   * @param {string} [opts.completeMsg=''] - ALERT message after last record completes
   * @param {Function} [opts.onEach] - Called with (recID, index, total) after each PUT
   * @example
   * req.batchUpdate(recordIDs, function(record) {
   *   record.record.form_values['c3d4'] = 'Batch Value'
   *   return record
   * }, { completeMsg: 'All records updated!' })
   */
  batchUpdate = (recIDs, buildFn, opts = {}) => {
    var total = recIDs.length;
    var completeMsg = opts.completeMsg || '';
    var self = this;

    for (var i = 0; i < total; i++) {
      (function(index) {
        var isLast = index === total - 1;
        self.modifyRecord(recIDs[index], buildFn, function() {
          if (opts.onEach) opts.onEach(recIDs[index], index, total);
          if (isLast && completeMsg) ALERT(completeMsg);
        });
      })(i);
    }
  };

  /**
   * @summary Generic external API request
   * @description For non-Fulcrum APIs (ArcGIS, NWS, BLM, etc.). Sends the request
   * with auth header and parses JSON response.
   *
   * @param {Object} options - REQUEST-compatible options (url, qs, accept, headers, etc.)
   * @param {Function} callback - Called with parsed JSON data
   * @param {Function} [onError] - Per-request error override
   * @example
   * req.external({
   *   url: 'https://api.weather.gov/points/43.0,-89.0',
   *   accept: 'application/geo+json'
   * }, function(data) {
   *   SETVALUE('city', data.properties.relativeLocation.properties.city)
   * })
   */
  external = (options, callback, onError) => {
    this._request({
      url: options.url,
      method: options.method,
      headers: options.headers,
      qs: options.qs,
      accept: options.accept,
      contentType: options.contentType,
      body: options.body,
      skipAuth: true,
      onError: onError,
      onSuccess: function(data) {
        if (callback) callback(data);
      }
    });
  };
}
//#endregion

export { MRequest };
