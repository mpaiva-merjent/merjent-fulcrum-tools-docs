/**
 * Reusable permission methods for MerjentApp templates.
 *
 * Methods for determining user permission levels and applying field restrictions.
 * Methods are bound to the MerjentApp instance via templates (see {@link DefaultTemplate}).
 *
 * **Usage:** These methods are not called directly. They are bound to app methods
 * by a template function:
 *
 * ```javascript
 * // In a template:
 * app.getPermissions = PermissionMixins.fourTier.bind(app);
 * app.baseUpdatePermission = PermissionMixins.standardUpdate.bind(app);
 * ```
 *
 * @namespace PermissionMixins
 * @see {@link MerjentApp} for the main class
 * @see {@link DefaultTemplate} for how methods are bound
 */
export const PermissionMixins = {
  /**
   * Four-tier permission system: Admin | Editor | User | Viewer
   *
   * @param {*} event - Fulcrum event object
   * @returns {string} 'Admin', 'Editor', 'User', or 'Viewer'
   * @memberof MerjentApp.PermissionMixins
   */
  fourTier(event){
    return this.adminNames.includes(USERFULLNAME()) || this.adminRoles.includes(ROLE()) ? 'Admin' :
      this.editorNames.includes(USERFULLNAME()) || this.editorRoles.includes(ROLE()) ? 'Editor' :
        this.userNames.includes(USERFULLNAME()) || this.userRoles.includes(ROLE()) ? 'User' :
          this.viewerNames.includes(USERFULLNAME()) || this.viewerRoles.includes(ROLE()) ? 'Viewer' :
            this.defaultPermissionLevel;
  },

  /**
   * Standard update permission restrictions based on user level and complete lock status
   *
   * @param {*} event - Fulcrum event object
   * @returns {void}
   * @memberof MerjentApp.PermissionMixins
   */
  standardUpdate(event){
    const permission = this.getPermissions(event);
    // const isCompleteLocked = this.getCompleteLockStatus(event);

    return permission == 'Admin' ? this.standardAdmin(event) :
      permission == 'Editor' ? this.standardEditor(event) :
        permission == 'User' ? this.standardUser(event) :
          this.standardViewer(event);
  },

  /**
   * Admin permission restrictions. All fields editable unless complete lock is on.
   * Status is always editable.
   *
   * @param {*} event - Fulcrum event object
   * @returns {void}
   * @memberof MerjentApp.PermissionMixins
   */
  standardAdmin(event) {
    const isCompleteLocked = this.getCompleteLockStatus(event);
    isCompleteLocked ? M.setAllRO(true, this.completeLockExclude) : M.setAllRO(false);
    SETSTATUSREADONLY(false);
  },

  /**
   * Editor permission restrictions. Fields editable unless complete lock is on.
   * Location editing disabled if disableLocationEdit is true.
   *
   * @param {*} event - Fulcrum event object
   * @returns {void}
   * @memberof MerjentApp.PermissionMixins
   */
  standardEditor(event) {
    const isCompleteLocked = this.getCompleteLockStatus(event);
    isCompleteLocked ? M.setAllRO(true, this.completeLockExclude) : M.setAllRO(null);
    if (this.disableLocationEdit) M.disableLocationEdit();
  },

  /**
   * User permission restrictions. Fields editable unless complete lock is on.
   * Location editing disabled if disableLocationEdit is true.
   *
   * @param {*} event - Fulcrum event object
   * @returns {void}
   * @memberof MerjentApp.PermissionMixins
   */
  standardUser(event) {
    const isCompleteLocked = this.getCompleteLockStatus(event);
    isCompleteLocked ? M.setAllRO(true) : M.setAllRO(null);
    if (this.disableLocationEdit) M.disableLocationEdit();
  },

  /**
   * Viewer permission restrictions. All fields read-only, location editing disabled,
   * status read-only.
   *
   * @param {*} event - Fulcrum event object
   * @returns {void}
   * @memberof MerjentApp.PermissionMixins
   */
  standardViewer(event) {
    M.setAllRO(true);
    M.disableLocationEdit();
    SETSTATUSREADONLY(true);
  },

  //       SETHIDDEN(this.dnReviewed, true);

  //     }
  //   }
  // },

    /**
   * Restrict a set of fields to users with allowed permission levels.
   * Hides/readonly-locks for disallowed users, restores for allowed users.
   *
   * @param {Object}   opts
   * @param {string[]} [opts.allowed=['Admin','Editor']] - permission levels granted access
   * @param {string[]} [opts.hide=[]]     - data_names to SETHIDDEN
   * @param {string[]} [opts.readonly=[]] - data_names to SETREADONLY
   * @param {*} event
   */
  restrictFieldsByRole({ allowed = ['Admin', 'Editor'], hide = [], readonly = [] } = {}, event) {
    const restrict = !allowed.includes(this.getPermissions(event));
    hide.forEach(dn => dn && SETHIDDEN(dn, restrict));
    readonly.forEach(dn => dn && SETREADONLY(dn, restrict));
  },

  /**
   * Hide coordinator initials and lock the review field for non-coordinators.
   * Thin wrapper over restrictFieldsByRole using the app's configured dn* props.
   */
  hideCoordInit(event) {
    this.restrictFieldsByRole({
      allowed: this.coordinatorRoles || ['Admin', 'Editor'],
      hide:     [this.dnCoordinatorInitials,this.dnCoordinatorInitials1,this.dnReviewed, this.dnReviewed1],
      readonly: [this.dnReviewed, this.dnReviewed1],
    }, event);
  }
};
